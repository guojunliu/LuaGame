
function start()
    local length = 5;
    local interval = 0.5;
    floorCubeSpArray = {}

    for i = 0, 20, 1 do
        for j = 0, 15, 1 do

            local cube = CS.UnityEngine.GameObject.CreatePrimitive(CS.UnityEngine.PrimitiveType.Cube);
            cube.transform.position = CS.UnityEngine.Vector3(i * (length + interval), 0, j * (length + interval));
            cube.transform.localScale = CS.UnityEngine.Vector3(length, 1, length);
            cube.transform:SetParent(self.transform);

            local floorSp = cube:AddComponent(typeof(CS.FloorCube));
            floorSp.column = i;
            floorSp.row = j;
            floorSp.gameobject = cube;

            floorCubeSpArray[(i-1)*10+j] = floorSp;
        end
    end

    print("floorCubeSpArray count : " .. #floorCubeSpArray);

    local aaaaaa =  CS.GameManager.Instance.aaa;

    print("dawdawda " .. aaaaaa);

    CS.GameManager.Instance.floorCubeSpArray = floorCubeSpArray;

    aa = CS.GameManager.Instance.floorCubeSpArray;

    --print("aaa : " .. aa);
    print("aaa count : " .. aa.Count)
end
